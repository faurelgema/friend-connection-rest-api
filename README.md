# GO Friends Connection REST API


made with GO, using gin-gonic/gin framework and gorm for the ORM

                    
Installation & Run
host     := "db"
port     := 5432
user     := "postgres"
password := "postgres"
dbname   := "friend-mgmt"

# USE THIS LINK AFTER RUNNING THE PROGRAM 
http://localhost:3000/swagger/index.html
